# harness_rules.py — supply-chain-agent용 자동 생성 # Optimize v1.0 · HAchillesWorld from hachillesworld.harness import HarnessSpec, Rule class SupplyChainHarness(HarnessSpec): """L2.3 → L3 전환을 위한 하네스 규칙""" def calibration_check(self, pred, actual): # ECE 0.22 → 0.10 미만 목표 (Scan §12 기반) ece = self._compute_ece(pred, actual) if ece > 0.10: return Rule.REJECT( reason=f"Calibration ECE {ece:.3f} > 0.10", action="recalibrate" ) return Rule.PASS() def self_correction_check(self, episode): # Self-Correction Rate 0% → 10% 이상 목표 if episode.has_error and not episode.self_corrected: return Rule.RETRY( prompt="이전 응답에서 오류를 감지했습니다. 다시 시도하세요.", max_attempts=3 ) return Rule.PASS() def harness_coverage_check(self, action): # 커버리지 3개 → 20개 목표 (자동 확장 중) return self._validate_action_bounds(action)
# cost_router.py — 예상 절감 $18,320/년 # CostAwareRouter for supply-chain-agent (L2.3) from hachillesworld.optimize import CostAwareRouter router = CostAwareRouter( agent_name="supply-chain-agent", # Scan §15: 복잡도 분포 분석 결과 반영 tiers=[ {"complexity": "low", "model": "claude-haiku-4-5", "threshold": 0.3}, {"complexity": "medium", "model": "claude-sonnet-4-6", "threshold": 0.7}, {"complexity": "high", "model": "claude-opus-4-8", "threshold": 1.0}, ], # 현재 분포: low 12% / medium 61% / high 27% # 목표 분포: low 35% / medium 55% / high 10% ) def route_and_call(task: str, context: dict): model = router.select(task, context) return model.invoke(task) # 예상 월 절감: $1,527
# mcts_planner.py — L3 목표 계획기 스캐폴드 # MCTS for supply-chain-agent (L2.3 → L3) from hachillesworld.optimize.mcts import MCTSPlanner planner = MCTSPlanner( agent_name="supply-chain-agent", # Scan §8: 계획 깊이 분석 기반 파라미터 n_simulations=50, # 현재 L2.3 기준 권장값 exploration_c=1.414, # UCB1 상수 max_depth=5, # 롤아웃 최대 깊이 rollout_policy="random", # Phase 2에서 "neural"로 업그레이드 ) def plan_action(state, goal, world_model): root = planner.build_tree(state, goal, world_model) best_action = planner.best_action(root) return best_action # Phase 3: Meta-Harness와 연동해 자기 수정
